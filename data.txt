import { Component } from "react";
import { Header } from "./Header";
import Footer from "./Footer";

export default class AddUser extends Component {
  addUser = (e) => {
    e.preventDefault();

    const data = e.target.elements.uname.value;
    this.props.au(data);
  };
  render() {
    return (
      <div>
        <form onSubmit={this.addUser}>
          <input type="text" name="uname" />
          <button>add</button>
        </form>
      </div>
    );
  }
}




import { Component } from "react";
import { Header } from "./Header";
import Footer from "./Footer";
import Users from "./Users";
import AddUser from "./AddUser";

export default class MainApp extends Component {
  state = {
    userData: [],
    headerData: "welcome to header",
  };
  addUser = (user) => {
    this.setState((prevSate) => {
      return {
        userData: prevSate.userData.concat(user),
      };
    });
  };

  render() {
    return (
      <div>
        <Header hdata={this.state.headerData} />
        <Users users={this.state.userData} />
        <p> MainApp Declared</p>
        <AddUser  au={this.addUser}/>
        <Footer />
      </div>
    );
  }
}
