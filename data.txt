import { Component } from "react";
import { Header } from "./Header";
import Footer from "./Footer";
import Users from "./Users";
import AddUser from "./AddUser";

export default class MainApp extends Component {
  state = {
    userData: [],
    headerData: "User CRUD Application",
  };
  addUser = (user) => {
    this.setState((prevSate) => {
      return {
        userData: prevSate.userData.concat(user),
      };
    });
  };
deleteUsers=()=>{
  this.setState(()=>{
    return{
      userData:[]
    }
  })
}
deleteUser=(user)=>{
  this.setState((prevSate)=>{
    return{
      userData:prevSate.userData.filter((option)=> user !== option)
    }
  })
}

  render() {
    return (
      <div>
        <Header hdata={this.state.headerData} 
        du={this.deleteUsers}
        hasData={this.state.userData.length > 0}
        
        />
        <Users users={this.state.userData} 
        du={this.deleteUser}
        />
         <AddUser  au={this.addUser}/>
        <Footer />
      </div>
    );
  }
}
