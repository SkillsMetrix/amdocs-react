componentDidMount() {
        const json = localStorage.getItem('amdocs')
        const userData = JSON.parse(json)
        if (userData) {
            this.setState(() => {
                return {
                    userData
                }
            })
        }
    }
      componentDidUpdate() {
        const json = JSON.stringify(this.state.userData)
        localStorage.setItem('amdocs', json)
    }
